import matplotlib.pyplot as plt

# 1. –ó–∞–ø–∏—Å—ã–≤–∞–µ–º —Å—é–¥–∞ —Å–≤–æ—ë —á–∏—Å–ª–æ a_1 –∏–∑ –∑–∞–¥–∞–Ω–∏—è ‚Ññ1.
# –ó–ê–ú–ï–ù–ò–¢–ï –∑–Ω–∞—á–µ–Ω–∏–µ —Å–ø—Ä–∞–≤–∞ –Ω–∞ –≤–∞—à–µ!
a_1 = 1.9587  # –ø—Ä–∏–º–µ—Ä. –ü–æ–¥—Å—Ç–∞–≤—å—Ç–µ –≤–∞—à–µ —á–∏—Å–ª–æ –∏–∑ ‚Ññ1.

# 2. –ú—ã —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞–µ–º —Å–µ–º–µ–π—Å—Ç–≤–æ —Ñ—É–Ω–∫—Ü–∏–π
#    h(x) = a_1 * x^3 + a_2 * x + a_1 * a_2
#    –ó–¥–µ—Å—å a_1 —Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–æ (–≤–∞—à–µ), –∞ a_2 –º—ã –ø–æ–¥–±–∏—Ä–∞–µ–º.

# 3. –¶–µ–ª–µ–≤—ã–µ —É—Å–ª–æ–≤–∏—è:

    a_2_candidate += step_a2
