import math
import numpy as np
import matplotlib.pyplot as plt

# 1. –ü–æ–¥—Å—Ç–∞–≤—å—Ç–µ —Å—é–¥–∞ —Å–≤–æ–∏ —á–∏—Å–ª–∞ a_1 –∏ a_2
#    (—Ç–æ, —á—Ç–æ –≤—ã –ø–æ–ª—É—á–∏–ª–∏ —Ä–∞–Ω–µ–µ).
#    –í–ê–ñ–ù–û: –∑–∞–º–µ–Ω–∏—Ç–µ –∑–Ω–∞—á–µ–Ω–∏—è —Å–ø—Ä–∞–≤–∞ –Ω–∞ –≤–∞—à–∏!
a_1 = 4.2331  # –ø—Ä–∏–º–µ—Ä –∏–∑ –∑–∞–¥–∞–Ω–∏—è ‚Ññ1
a_2 = 1.2432  # –ø—Ä–∏–º–µ—Ä –∏–∑ –∑–∞–¥–∞–Ω–∏—è ‚Ññ2

# 2. –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ñ—É–Ω–∫—Ü–∏—é r(x) = a_1*sin(x) + a_2*cos(2*x).
#    –°–Ω–∞—á–∞–ª–∞ –≤–µ—Ä—Å–∏—è –¥–ª—è –æ–¥–Ω–æ–≥–æ x (—Å–∫–∞–ª—è—Ä):
def r_scalar(x):
    return a_1*math.sin(x) + a_2*math.cos(2*x)

#    –¢–µ–ø–µ—Ä—å –≤–µ–∫—Ç–æ—Ä–Ω–∞—è –≤–µ—Ä—Å–∏—è –¥–ª—è –º–∞—Å—Å–∏–≤–∞ xs (numpy):
def r_np(xarr):
    return a_1*np.sin(xarr) + a_2*np.cos(2*xarr)

# 3. –î–µ–ª–∞–µ–º —Å–µ—Ç–∫—É —Ç–æ—á–µ–∫ –Ω–∞ [0; 2*pi]
xs = np.linspace(0, 2*math.pi, 400)

# 4. –°—á–∏—Ç–∞–µ–º –∑–Ω–∞—á–µ–Ω–∏—è —Ñ—É–Ω–∫—Ü–∏–∏ r(x) –¥–ª—è –≤—Å–µ—Ö —Ç–æ—á–µ–∫ xs
rs = r_np(xs)

# 5. –°—Ä–µ–¥–Ω–µ–µ –∑–Ω–∞—á–µ–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏ –Ω–∞ —ç—Ç–æ–º –ø—Ä–æ–º–µ–∂—É—Ç–∫–µ
m = float(np.mean(rs))

# 6. –û—Ü–µ–Ω–∏–≤–∞–µ–º "—Å—Ä–µ–¥–Ω–∏–π —Ä–∞–∑–±—Ä–æ—Å" –≤–æ–∫—Ä—É–≥ —Å—Ä–µ–¥–Ω–µ–≥–æ.
#    –≠—Ç–æ –ø–æ —Å–≤–æ–µ–π —Å—É—Ç–∏ –∏ –µ—Å—Ç—å —Å—Ä–µ–¥–Ω–µ–∫–≤–∞–¥—Ä–∞—Ç–∏—á–Ω–æ–µ –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏–µ.
sq = 0.0
for v in rs:
    sq += (v - m)**2
std_val = math.sqrt(sq / len(rs))

# 7. –û–∫—Ä—É–≥–ª—è–µ–º —Ä–∞–∑–±—Ä–æ—Å –¥–æ 4 –∑–Ω–∞–∫–æ–≤ –ø–æ—Å–ª–µ –∑–∞–ø—è—Ç–æ–π ‚Äî —ç—Ç–æ –∏ –µ—Å—Ç—å a_3.
a_3 = round(std_val, 4)

print("–í–ê–®–ï –ß–ò–°–õ–û a_3 =", f"{a_3:.4f}")
print("–°—Ä–µ–¥–Ω–µ–∫–≤–∞–¥—Ä–∞—Ç–∏—á–Ω–æ–µ –æ—Ç–∫–ª–æ–Ω–µ–Ω–∏–µ =", f"{std_val:.6f}")

# 8. –†–∏—Å—É–µ–º –≥—Ä–∞—Ñ–∏–∫ r(x) –∏ –≥–æ—Ä–∏–∑–æ–Ω—Ç–∞–ª—å–Ω—ã–µ –ª–∏–Ω–∏–∏ m, m ¬± std_val.
plt.figure(figsize=(6, 2.4))
plt.plot(xs, rs, lw=2)

# –ª–∏–Ω–∏—è —Å—Ä–µ–¥–Ω–µ–≥–æ

# –≥—Ä–∞–Ω–∏—Ü—ã "–∫–æ—Ä–∏–¥–æ—Ä–∞ –∫–æ–ª–µ–±–∞–Ω–∏–π"
