import math
import numpy as np
import matplotlib.pyplot as plt

# 1. –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ñ—É–Ω–∫—Ü–∏—é f(x).
def f(x):
    # f(x) = sqrt(x+1) + 0.5 * sin(3*x)
    # sqrt(x+1) –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∞, –µ—Å–ª–∏ x >= -1.
    # –ù–∞–º –Ω—É–∂–Ω–∞ —Ç–æ—á–∫–∞ x = 1, —ç—Ç–æ –±–µ–∑–æ–ø–∞—Å–Ω–æ.
    return math.sqrt(x + 1.0) + 0.5 * math.sin(3.0 * x)

# 2. –¶–µ–ª–µ–≤–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ, –≤ –∫–æ—Ç–æ—Ä–æ–µ —Ö–æ—Ç–∏–º –ø–æ–ø–∞—Å—Ç—å –ø—Ä–∏ x = 1.
y_star = 5.0

# 3. –ó–∞—Ä–∞–Ω–µ–µ —Å—á–∏—Ç–∞–µ–º f(1).
x0 = 1.0
f_x0 = f(x0)

# 4. –ü–µ—Ä–µ–±–æ—Ä –∑–Ω–∞—á–µ–Ω–∏–π a.
a_left  = -5.0
a_right =  5.0
step_a  =  0.001

best_a   = None
best_err = 10**9  # –æ—á–µ–Ω—å –±–æ–ª—å—à–æ–µ —Å—Ç–∞—Ä—Ç–æ–≤–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ

a = a_left
while a <= a_right + 1e-12:
    # g(1) = a * f(1) + a**2
    g_x0 = a * f_x0 + a**2

    # –æ—à–∏–±–∫–∞ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ —Ü–µ–ª–µ–≤–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è y_star
    err = abs(g_x0 - y_star)

    # –µ—Å–ª–∏ —Ç–µ–∫—É—â–µ–µ –ª—É—á—à–µ ‚Äî –æ–±–Ω–æ–≤–ª—è–µ–º
    if err < best_err:
        best_err = err
        best_a   = a

    a += step_a

# 5. –§–æ—Ä–º–∏—Ä—É–µ–º —á–∏—Å–ª–æ a_1 –∏ –ø–µ—á–∞—Ç–∞–µ–º.
a_1 = round(best_a, 4)

print("–í–ê–®–ï –ß–ò–°–õ–û a_1 =", f"{a_1:.4f}")
print("–ú–∏–Ω–∏–º–∞–ª—å–Ω–∞—è –æ—à–∏–±–∫–∞ =", f"{best_err:.6f}")

# 6. –ù–µ–±–æ–ª—å—à–æ–π –≥—Ä–∞—Ñ–∏–∫ g(x) = a_1 * f(x) + a_1**2 –Ω–∞ [-2; 2].
def f_np(xarr):
    return np.sqrt(xarr + 1.0) + 0.5 * np.sin(3.0 * xarr)

xs = np.linspace(-2, 2, 400)
g_vals = a_1 * f_np(xs) + a_1**2

plt.figure(figsize=(6, 2.4))
plt.plot(xs, g_vals, lw=2)
plt.axvline(x0, color="#999", lw=1, alpha=.6)  # –æ—Ç–º–µ—á–∞–µ–º x = 1
plt.title("g(x) –ø—Ä–∏ –Ω–∞–π–¥–µ–Ω–Ω–æ–º a_1")
plt.grid(True, alpha=.3)
