import math
import numpy as np
import matplotlib.pyplot as plt

# 1. –ü–æ–¥—Å—Ç–∞–≤—å—Ç–µ —Å—é–¥–∞ —Å–≤–æ–∏ —á–∏—Å–ª–∞ a_1, a_2 –∏ a_3,
#    –∫–æ—Ç–æ—Ä—ã–µ –≤—ã —É–∂–µ –Ω–∞—à–ª–∏ —Ä–∞–Ω–µ–µ.
#    –í–ê–ñ–ù–û: –∑–∞–º–µ–Ω–∏—Ç–µ –∑–Ω–∞—á–µ–Ω–∏—è —Å–ø—Ä–∞–≤–∞ –Ω–∞ –≤–∞—à–∏!
a_1 = 6.1847  # –ø—Ä–∏–º–µ—Ä –∏–∑ –∑–∞–¥–∞–Ω–∏—è ‚Ññ1
a_2 = 2.2235  # –ø—Ä–∏–º–µ—Ä –∏–∑ –∑–∞–¥–∞–Ω–∏—è ‚Ññ2
a_3 = 3.0598  # –ø—Ä–∏–º–µ—Ä –∏–∑ –∑–∞–¥–∞–Ω–∏—è ‚Ññ3

# 2. –ò—Å—Ö–æ–¥–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è r(x) = a_1*sin(x) + a_2*cos(2*x).
#    –°–Ω–∞—á–∞–ª–∞ –≤–µ—Ä—Å–∏—è –¥–ª—è –æ–¥–Ω–æ–≥–æ x (—Å–∫–∞–ª—è—Ä):
def r_scalar(x):
    return a_1*math.sin(x) + a_2*math.cos(2*x)

#    –í–µ–∫—Ç–æ—Ä–Ω–∞—è –≤–µ—Ä—Å–∏—è –¥–ª—è –º–∞—Å—Å–∏–≤–∞ xs (numpy):
def r_np(xarr):
    return a_1*np.sin(xarr) + a_2*np.cos(2*xarr)

# 3. –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è s(x).
#    s(x) = (1 + 0.1*a_1) * r(x - a_2) + a_3
#    –≠—Ç–æ:
#    - —Å–¥–≤–∏–≥ –ø–æ x –Ω–∞ a_2,
#    - —Ä–∞—Å—Ç—è–∂–µ–Ω–∏–µ –ø–æ y –≤ (1 + 0.1*a_1) —Ä–∞–∑,
#    - —Å–¥–≤–∏–≥ –≥—Ä–∞—Ñ–∏–∫–∞ –≤–≤–µ—Ä—Ö –Ω–∞ a_3.
def s_scalar(x):
    return (1 + 0.1*a_1)*r_scalar(x - a_2) + a_3

def s_np(xarr):
    return (1 + 0.1*a_1)*r_np(xarr - a_2) + a_3

# 4. –î–µ–ª–∞–µ–º —Å–µ—Ç–∫—É –ø–æ [0; 2*pi].
xs = np.linspace(0, 2*math.pi, 400)

# 5. –°—á–∏—Ç–∞–µ–º –∑–Ω–∞—á–µ–Ω–∏—è –æ–±–µ–∏—Ö —Ñ—É–Ω–∫—Ü–∏–π –Ω–∞ —Å–µ—Ç–∫–µ.
rs = r_np(xs)
ss = s_np(xs)

# 6. –ò—â–µ–º –º–∞–∫—Å–∏–º—É–º –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ s(x).
max_val = float(np.max(ss))

# 7. –§–æ—Ä–º–∏—Ä—É–µ–º —á–∏—Å–ª–æ a_4 –∏ –ø–µ—á–∞—Ç–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç.
a_4 = round(max_val, 4)

print("–ú–∞–∫—Å–∏–º—É–º –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ =", f"{max_val:.6f}")

# 8. –†–∏—Å—É–µ–º –æ–±–µ —Ñ—É–Ω–∫—Ü–∏–∏ –Ω–∞ –æ–¥–Ω–æ–º –≥—Ä–∞—Ñ–∏–∫–µ,
#    —á—Ç–æ–±—ã —É–≤–∏–¥–µ—Ç—å —Å–¥–≤–∏–≥, —Ä–∞—Å—Ç—è–∂–µ–Ω–∏–µ –∏ –ø–æ–¥—ä—ë–º.

# –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–Ω–∞—è s(x): –¥–µ–ª–∞–µ–º –ª–∏–Ω–∏—é –ø–æ—Ç–æ–ª—â–µ
