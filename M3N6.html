#    –∫–æ—Ç–æ—Ä—ã–µ –≤—ã –ø–æ–ª—É—á–∏–ª–∏ —Ä–∞–Ω–µ–µ.
#    –í–ê–ñ–ù–û: –∑–∞–º–µ–Ω–∏—Ç–µ –∑–Ω–∞—á–µ–Ω–∏—è —Å–ø—Ä–∞–≤–∞ –Ω–∞ –≤–∞—à–∏!
a_1 = 9.1049   # –ø—Ä–∏–º–µ—Ä –∏–∑ –∑–∞–¥–∞–Ω–∏—è ‚Ññ1
a_2 = 3.9587   # –ø—Ä–∏–º–µ—Ä –∏–∑ –∑–∞–¥–∞–Ω–∏—è ‚Ññ2
a_3 = 4.0489   # –ø—Ä–∏–º–µ—Ä –∏–∑ –∑–∞–¥–∞–Ω–∏—è ‚Ññ3
a_4 = 5.0948   # –ø—Ä–∏–º–µ—Ä –∏–∑ –∑–∞–¥–∞–Ω–∏—è ‚Ññ4
a_5 = 98.8947  # –ø—Ä–∏–º–µ—Ä –∏–∑ –∑–∞–¥–∞–Ω–∏—è ‚Ññ5 (–≤ –ø—Ä–æ—Ü–µ–Ω—Ç–∞—Ö)

# 2. –û–ø—Ä–µ–¥–µ–ª—è–µ–º –∫—É—Å–æ—á–Ω–æ-–∑–∞–¥–∞–Ω–Ω—É—é —Ñ—É–Ω–∫—Ü–∏—é t(x).
#    t(x) = a_1*x + a_2,            –µ—Å–ª–∏ x < -1

def t_scalar(x):
    if x < -1:
        return a_1*x + a_2
    elif x <= 1:
        return -a_3*(x**2) + a_4
    else:
        return a_4 + a_5/10

#    –í–µ—Ä—Å–∏—è –¥–ª—è –º–∞—Å—Å–∏–≤–∞ xs (numpy) ‚Äî –ø—Ä–∏–≥–æ–¥–∏—Ç—Å—è –Ω–∏–∂–µ –¥–ª—è —Å—Ä–µ–¥–Ω–µ–≥–æ:
def t_np(xarr):
    vals = []
    for x in xarr:
        vals.append(t_scalar(x))
    return np.array(vals)

# 3. –ì–æ—Ç–æ–≤–∏–º —Ç–æ—á–∫–∏ —Å–ø–µ—Ü–∏–∞–ª—å–Ω–æ –¥–ª—è –ì–†–ê–§–ò–ö–ê,
#    —á—Ç–æ–±—ã –Ω–µ –±—ã–ª–æ "—Å–æ–µ–¥–∏–Ω–µ–Ω–∏—è —á–µ—Ä–µ–∑ —Ä–∞–∑—Ä—ã–≤".
#
#    –õ–µ–≤—ã–π –∫—É—Å–æ–∫: x < -1  (–∏—Å–∫–ª—é—á–∞–µ–º —Å–∞–º—É -1)
xs_left = np.linspace(-3, -1, 200, endpoint=False)
ys_left = a_1*xs_left + a_2

#    –°—Ä–µ–¥–Ω–∏–π –∫—É—Å–æ–∫: -1 <= x <= 1  (–≤–∫–ª—é—á–∞–µ–º –∏ -1, –∏ 1)
xs_mid = np.linspace(-1, 1, 200)
ys_mid = -a_3*(xs_mid**2) + a_4

#    –ü—Ä–∞–≤—ã–π –∫—É—Å–æ–∫: x > 1  (–∏—Å–∫–ª—é—á–∞–µ–º —Å–∞–º—É 1)
xs_right = np.linspace(1, 3, 200, endpoint=False)
ys_right = (a_4 + a_5/10) * np.ones_like(xs_right)

# 4. –†–∏—Å—É–µ–º –≥—Ä–∞—Ñ–∏–∫ –∫—É—Å–æ—á–Ω–æ-–∑–∞–¥–∞–Ω–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ (–∫–∞–∂–¥—ã–π –∫—É—Å–æ–∫ –æ—Ç–¥–µ–ª—å–Ω–æ).
plt.figure(figsize=(6, 2.4))

plt.plot(xs_left,  ys_left,  lw=2)
plt.plot(xs_mid,   ys_mid,   lw=2)
plt.plot(xs_right, ys_right, lw=2)

#    –¢–µ–ø–µ—Ä—å –∞–∫–∫—É—Ä–∞—Ç–Ω–æ –æ—Ç–º–µ—á–∞–µ–º —Ç–æ—á–∫–∏ —Å—Ç—ã–∫–∞ –ø—Ä–∞–≤–∏–ª –≤ x = -1 –∏ x = 1.
#    –ò–¥–µ—è:
#    - –≤—ã–∫–æ–ª–æ—Ç—ã–π (–ø—É—Å—Ç–æ–π) –∫—Ä—É–∂–æ–∫ —Ç–∞–º, –≥–¥–µ —Ç–æ—á–∫–∞ –ù–ï –≤—Ö–æ–¥–∏—Ç –≤ —ç—Ç–æ—Ç –∫—É—Å–æ–∫;
#    - –∑–∞–∫—Ä–∞—à–µ–Ω–Ω—ã–π –∫—Ä—É–∂–æ–∫ —Ç–∞–º, –≥–¥–µ —Ç–æ—á–∫–∞ –≤—Ö–æ–¥–∏—Ç.

#    x = -1:
xL = -1.0
y_left_at_L   = a_1*xL + a_2           # –ª–µ–≤—ã–π –∫—É—Å–æ–∫ (x < -1), —Ç–æ—á–∫–∞ -1 –ù–ï –≤—Ö–æ–¥–∏—Ç
y_middle_at_L = -a_3*(xL**2) + a_4     # —Å—Ä–µ–¥–Ω–∏–π –∫—É—Å–æ–∫ (-1 <= x <= 1), —Ç–æ—á–∫–∞ -1 –í–•–û–î–ò–¢

#    –ü—É—Å—Ç–æ–π –∫—Ä—É–∂–æ–∫ –¥–ª—è –ª–µ–≤–æ–≥–æ –∫—É—Å–∫–∞ –ø—Ä–∏ x = -1:
plt.scatter([xL], [y_left_at_L],
            s=40,
            facecolors='none',
            edgecolors='black')

#    –ó–∞–∫—Ä–∞—à–µ–Ω–Ω—ã–π –∫—Ä—É–∂–æ–∫ –¥–ª—è —Å—Ä–µ–¥–Ω–µ–≥–æ –∫—É—Å–∫–∞ –ø—Ä–∏ x = -1:
plt.scatter([xL], [y_middle_at_L],
            s=40,
            color='black')

#    x = 1:
xR = 1.0
y_middle_at_R = -a_3*(xR**2) + a_4     # —Å—Ä–µ–¥–Ω–∏–π –∫—É—Å–æ–∫ (-1 <= x <= 1), —Ç–æ—á–∫–∞ 1 –í–•–û–î–ò–¢
y_right_at_R  = a_4 + a_5/10           # –ø—Ä–∞–≤—ã–π –∫—É—Å–æ–∫ (x > 1), —Ç–æ—á–∫–∞ 1 –ù–ï –≤—Ö–æ–¥–∏—Ç

#    –ó–∞–∫—Ä–∞—à–µ–Ω–Ω—ã–π –∫—Ä—É–∂–æ–∫ –¥–ª—è —Å—Ä–µ–¥–Ω–µ–≥–æ –∫—É—Å–∫–∞ –ø—Ä–∏ x = 1:
plt.scatter([xR], [y_middle_at_R],
            s=40,
            color='black')

#    –ü—É—Å—Ç–æ–π –∫—Ä—É–∂–æ–∫ –¥–ª—è –ø—Ä–∞–≤–æ–≥–æ –∫—É—Å–∫–∞ –ø—Ä–∏ x = 1:
plt.scatter([xR], [y_right_at_R],
            s=40,
            facecolors='none',
            edgecolors='black')

plt.xlabel("–æ—Å—å x")
plt.ylabel("t(x)")
plt.title("–ö—É—Å–æ—á–Ω–æ-–∑–∞–¥–∞–Ω–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è t(x)")
plt.grid(True, alpha=.3)

# 5. –°—á–∏—Ç–∞–µ–º —Å—Ä–µ–¥–Ω–µ–µ –∑–Ω–∞—á–µ–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏ –Ω–∞ [-2; 2].
#    –ó–¥–µ—Å—å –Ω–∞–º —É–¥–æ–±–Ω–æ –≤–∑—è—Ç—å —Å–ø–ª–æ—à–Ω—É—é —Å–µ—Ç–∫—É –∏ –≤–æ—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å—Å—è t_np.
xs_full = np.linspace(-3, 3, 600)
ts_full = t_np(xs_full)

mask_mid = (xs_full >= -2) & (xs_full <= 2)
ts_mid = ts_full[mask_mid]

mean_val = float(np.mean(ts_mid))

# 6. –§–æ—Ä–º–∏—Ä—É–µ–º —á–∏—Å–ª–æ a_6 –∏ –ø–µ—á–∞—Ç–∞–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç.
a_6 = round(mean_val, 4)

print("–í–ê–®–ï –ß–ò–°–õ–û a_6 =", f"{a_6:.4f}")
print("–°—Ä–µ–¥–Ω–µ–µ –∑–Ω–∞—á–µ–Ω–∏–µ t(x) –Ω–∞ [-2; 2] =", f"{mean_val:.6f}")
